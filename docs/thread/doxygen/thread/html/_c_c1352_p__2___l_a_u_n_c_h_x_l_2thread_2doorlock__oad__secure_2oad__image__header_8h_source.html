171 /*
172  * The complete image header is a combination of:
173  * First: struct img_common_header
174  * Followed by 1 or more, in any order:
175  * struct img_seg_payload
176  * struct img_seg_boundary
177  * struct img_seg_signature
178  * The last item is always:
179  * struct img_seg_last
180  *
181  * A typical true header might be as follows
182  * struct my_header {
183  * struct img_common_header header;
184  * struct img_seg_payload payload;
185  * struct img_seg_last last;
186  * };
187  *
188  * Or
189  * struct my_header {
190  * struct img_common_header header;
191  * struct img_seg_boundary boundary;
192  * struct img_seg_payload payload;
193  * struct img_seg_security security;
194  * struct img_seg_last last;
195  * };
196  */
