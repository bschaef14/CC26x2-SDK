185 private:
186  /*
187  * The RssAverager uses an adaptive exponentially weighted filter to maintain the average. It keeps track of
188  * current average and the number of added RSS values (up to a 8).
189  *
190  * For the first 8 added RSS values, the average is the arithmetic mean of the added values (e.g., if 5 values are
191  * added, the average is sum of the 5 added RSS values divided by 5. After the 8th RSS value, a weighted filter is
192  * used with coefficients (1/8, 7/8), i.e., newAverage = 1/8 * newRss + 7/8 * oldAverage.
193  *
194  * To add to accuracy of the averaging process, the RSS values and the maintained average are multiplied by a
195  * precision factor of -8.
196  *
197  */
198 
199  enum
200  {
201  kPrecisionBitShift = 3, // Precision multiple for RSS average (1 << PrecisionBitShift).
202  kPrecision = (1 << kPrecisionBitShift),
203  kPrecisionBitMask = (kPrecision - 1),
204 
205  kCoeffBitShift = 3, // Coefficient used for exponentially weighted filter (1 << kCoeffBitShift).
206  };
207 
208  // Member variables fit into two bytes.
209 
210  uint16_t mAverage : 11; // The raw average signal strength value (stored as RSS times precision multiple).
211  uint16_t mCount : 5; // Number of RSS values added to averager so far (limited to 2^kCoeffBitShift-1).
212 };
