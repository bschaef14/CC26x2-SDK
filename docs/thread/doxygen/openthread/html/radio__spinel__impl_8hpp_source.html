254 template <typename InterfaceType, typename ProcessContextType>
255 bool RadioSpinel<InterfaceType, ProcessContextType>::IsRcp(void)
256 {
257  uint8_t capsBuffer[kCapsBufferSize];
258  const uint8_t *capsData = capsBuffer;
259  spinel_size_t capsLength = sizeof(capsBuffer);
260  bool supportsRawRadio = false;
261  bool isRcp = false;
262 
263  SuccessOrDie(Get(SPINEL_PROP_CAPS, SPINEL_DATATYPE_DATA_S, capsBuffer, &capsLength));
264 
265  while (capsLength > 0)
266  {
267  unsigned int capability;
268  spinel_ssize_t unpacked;
269 
270  unpacked = spinel_datatype_unpack(capsData, capsLength, SPINEL_DATATYPE_UINT_PACKED_S, &capability);
271  VerifyOrDie(unpacked > 0, OT_EXIT_RADIO_SPINEL_INCOMPATIBLE);
272 
273  if (capability == SPINEL_CAP_MAC_RAW)
274  {
275  supportsRawRadio = true;
276  }
277 
278  if (capability == SPINEL_CAP_CONFIG_RADIO)
279  {
280  isRcp = true;
281  }
282 
283  capsData += unpacked;
284  capsLength -= static_cast<spinel_size_t>(unpacked);
285  }
286 
287  if (!supportsRawRadio && isRcp)
288  {
289  otLogCritPlat("RCP capability list does not include support for radio/raw mode");
290  DieNow(OT_EXIT_RADIO_SPINEL_INCOMPATIBLE);
291  }
292 
293  return isRcp;
294 }

304  uint8_t capsBuffer[kCapsBufferSize];
305  const uint8_t *capsData = capsBuffer;
306  spinel_size_t capsLength = sizeof(capsBuffer);
307 
308  SuccessOrExit(error = Get(SPINEL_PROP_RADIO_CAPS, SPINEL_DATATYPE_UINT_PACKED_S, &radioCaps));
309  mRadioCaps = static_cast<otRadioCaps>(radioCaps);
310 
311  if ((mRadioCaps & kRequiredRadioCaps) != kRequiredRadioCaps)
312  {
313  otLogCritPlat("RCP does not support required capabilities: ack-timeout:%s, tx-retries:%s, CSMA-backoff:%s",
314  (mRadioCaps & OT_RADIO_CAPS_ACK_TIMEOUT) ? "yes" : "no",
315  (mRadioCaps & OT_RADIO_CAPS_TRANSMIT_RETRIES) ? "yes" : "no",
316  (mRadioCaps & OT_RADIO_CAPS_CSMA_BACKOFF) ? "yes" : "no");
317 
318  DieNow(OT_EXIT_RADIO_SPINEL_INCOMPATIBLE);
319  }
320 
321  SuccessOrExit(error = Get(SPINEL_PROP_CAPS, SPINEL_DATATYPE_DATA_S, capsBuffer, &capsLength));
322  while (capsLength > 0)
323  {
324  unsigned int capability;
325  spinel_ssize_t unpacked =
326  spinel_datatype_unpack(capsData, capsLength, SPINEL_DATATYPE_UINT_PACKED_S, &capability);
327 
328  VerifyOrDie(unpacked > 0, OT_EXIT_RADIO_SPINEL_INCOMPATIBLE);
329 
330  if (capability == SPINEL_CAP_OPENTHREAD_LOG_METADATA)
331  {
332  mSupportsLogStream = true;
333  }
334 
335  capsData += unpacked;
336  capsLength -= static_cast<spinel_size_t>(unpacked);
337  }
338 
339 exit:
340  return error;

371 {
372  otError error = OT_ERROR_NONE;
373  uint8_t header;

400 template <typename InterfaceType, typename ProcessContextType>
401 void RadioSpinel<InterfaceType, ProcessContextType>::HandleNotification(SpinelInterface::RxFrameBuffer &aFrameBuffer)
402 {
403  spinel_prop_key_t key;
404  spinel_size_t len = 0;
405  spinel_ssize_t unpacked;
406  uint8_t * data = NULL;
407  uint32_t cmd;
408  uint8_t header;
409  otError error = OT_ERROR_NONE;
410  bool shouldSaveFrame = false;
411 
412  unpacked = spinel_datatype_unpack(aFrameBuffer.GetFrame(), aFrameBuffer.GetLength(), "CiiD", &header, &cmd, &key,
413  &data, &len);
414  VerifyOrExit(unpacked > 0, error = OT_ERROR_PARSE);
415  VerifyOrExit(SPINEL_HEADER_GET_TID(header) == 0, error = OT_ERROR_PARSE);
416 
417  switch (cmd)
418  {
419  case SPINEL_CMD_PROP_VALUE_IS:
420  // Some spinel properties cannot be handled during `WaitResponse()`, we must cache these events.
421  // `mWaitingTid` is released immediately after received the response. And `mWaitingKey` is be set
422  // to `SPINEL_PROP_LAST_STATUS` at the end of `WaitResponse()`.
423 
424  if (!IsSafeToHandleNow(key))
425  {
426  ExitNow(shouldSaveFrame = true);
427  }
428 
429  HandleValueIs(key, data, static_cast<uint16_t>(len));
430  break;
431 
432  case SPINEL_CMD_PROP_VALUE_INSERTED:
433  case SPINEL_CMD_PROP_VALUE_REMOVED:
434  otLogInfoPlat("Ignored command %d", cmd);
435  break;
436 
437  default:
438  ExitNow(error = OT_ERROR_PARSE);
439  }
440 
441 exit:
442  if (shouldSaveFrame)
443  {
444  aFrameBuffer.SaveFrame();
445  }
446  else
447  {
448  aFrameBuffer.DiscardFrame();
449  }
450 
451  LogIfFail("Error processing notification", error);
452 }

475 template <typename InterfaceType, typename ProcessContextType>
476 void RadioSpinel<InterfaceType, ProcessContextType>::HandleResponse(const uint8_t *aBuffer, uint16_t aLength)
477 {
478  spinel_prop_key_t key;
479  uint8_t * data = NULL;
480  spinel_size_t len = 0;
481  uint8_t header = 0;
482  uint32_t cmd = 0;
483  spinel_ssize_t rval = 0;
484  otError error = OT_ERROR_NONE;
485 
486  rval = spinel_datatype_unpack(aBuffer, aLength, "CiiD", &header, &cmd, &key, &data, &len);
487  VerifyOrExit(rval > 0 && cmd >= SPINEL_CMD_PROP_VALUE_IS && cmd <= SPINEL_CMD_PROP_VALUE_REMOVED,
488  error = OT_ERROR_PARSE);
489 
490  if (mWaitingTid == SPINEL_HEADER_GET_TID(header))
491  {
492  HandleWaitingResponse(cmd, key, data, static_cast<uint16_t>(len));
493  FreeTid(mWaitingTid);
494  mWaitingTid = 0;
495  }
496  else if (mTxRadioTid == SPINEL_HEADER_GET_TID(header))
497  {
498  if (mState == kStateTransmitting)
499  {
500  HandleTransmitDone(cmd, key, data, static_cast<uint16_t>(len));
501  }
502 
503  FreeTid(mTxRadioTid);
504  mTxRadioTid = 0;
505  }
506  else
507  {
508  otLogWarnPlat("Unexpected Spinel transaction message: %u", SPINEL_HEADER_GET_TID(header));
509  error = OT_ERROR_DROP;
510  }
511 
512 exit:
513  LogIfFail("Error processing response", error);
514 }

516 template <typename InterfaceType, typename ProcessContextType>
517 otError RadioSpinel<InterfaceType, ProcessContextType>::ThreadDatasetHandler(const uint8_t *aBuffer, uint16_t aLength)
518 {
519  otError error = OT_ERROR_NONE;
520  otOperationalDataset opDataset;
521  bool isActive = ((mWaitingKey == SPINEL_PROP_THREAD_ACTIVE_DATASET) ? true : false);
522  Spinel::Decoder decoder;
523  MeshCoP::Dataset dataset(isActive ? MeshCoP::Dataset::kActive : MeshCoP::Dataset::kPending);
524 
525  memset(&opDataset, 0, sizeof(otOperationalDataset));
526  decoder.Init(aBuffer, aLength);
527 
528  while (!decoder.IsAllReadInStruct())
529  {
530  unsigned int propKey;
531 
532  SuccessOrExit(error = decoder.OpenStruct());
533  SuccessOrExit(error = decoder.ReadUintPacked(propKey));
534 
535  switch (static_cast<spinel_prop_key_t>(propKey))
536  {
537  case SPINEL_PROP_NET_MASTER_KEY:
538  {
539  const uint8_t *key;
540  uint16_t len;
541 
542  SuccessOrExit(error = decoder.ReadData(key, len));
543  VerifyOrExit(len == OT_MASTER_KEY_SIZE, error = OT_ERROR_INVALID_ARGS);
544  memcpy(opDataset.mMasterKey.m8, key, len);
545  opDataset.mComponents.mIsMasterKeyPresent = true;
546  break;
547  }
548 
549  case SPINEL_PROP_NET_NETWORK_NAME:
550  {
551  const char *name;
552  size_t len;
553 
554  SuccessOrExit(error = decoder.ReadUtf8(name));
555  len = StringLength(name, OT_NETWORK_NAME_MAX_SIZE);
556  memcpy(opDataset.mNetworkName.m8, name, len);
557  opDataset.mNetworkName.m8[len] = '\0';
558  opDataset.mComponents.mIsNetworkNamePresent = true;
559  break;
560  }
561 
562  case SPINEL_PROP_NET_XPANID:
563  {
564  const uint8_t *xpanid;
565  uint16_t len;
566 
567  SuccessOrExit(error = decoder.ReadData(xpanid, len));
568  VerifyOrExit(len == OT_EXT_PAN_ID_SIZE, error = OT_ERROR_INVALID_ARGS);
569  memcpy(opDataset.mExtendedPanId.m8, xpanid, len);
570  opDataset.mComponents.mIsExtendedPanIdPresent = true;
571  break;
572  }
573 
574  case SPINEL_PROP_IPV6_ML_PREFIX:
575  {
576  const otIp6Address *addr;
577  uint8_t prefixLen;
578 
579  SuccessOrExit(error = decoder.ReadIp6Address(addr));

581  VerifyOrExit(prefixLen == OT_IP6_PREFIX_BITSIZE, error = OT_ERROR_INVALID_ARGS);
582  memcpy(opDataset.mMeshLocalPrefix.m8, addr, OT_MESH_LOCAL_PREFIX_SIZE);
583  opDataset.mComponents.mIsMeshLocalPrefixPresent = true;
584  break;
585  }
586 
587  case SPINEL_PROP_DATASET_DELAY_TIMER:
588  {
589  SuccessOrExit(error = decoder.ReadUint32(opDataset.mDelay));
590  opDataset.mComponents.mIsDelayPresent = true;
591  break;
592  }
593 
594  case SPINEL_PROP_MAC_15_4_PANID:
595  {
596  SuccessOrExit(error = decoder.ReadUint16(opDataset.mPanId));
597  opDataset.mComponents.mIsPanIdPresent = true;
598  break;
599  }
600 
601  case SPINEL_PROP_PHY_CHAN:
602  {
603  uint8_t channel;
604 
605  SuccessOrExit(error = decoder.ReadUint8(channel));
606  opDataset.mChannel = channel;
607  opDataset.mComponents.mIsChannelPresent = true;
608  break;
609  }
610 
611  case SPINEL_PROP_NET_PSKC:
612  {
613  const uint8_t *psk;
614  uint16_t len;
615 
616  SuccessOrExit(error = decoder.ReadData(psk, len));
617  VerifyOrExit(len == OT_PSKC_MAX_SIZE, error = OT_ERROR_INVALID_ARGS);
618  memcpy(opDataset.mPskc.m8, psk, OT_PSKC_MAX_SIZE);
619  opDataset.mComponents.mIsPskcPresent = true;
620  break;
621  }
622 
623  case SPINEL_PROP_DATASET_SECURITY_POLICY:
624  {
625  SuccessOrExit(error = decoder.ReadUint16(opDataset.mSecurityPolicy.mRotationTime));
626  SuccessOrExit(error = decoder.ReadUint8(opDataset.mSecurityPolicy.mFlags));
627  opDataset.mComponents.mIsSecurityPolicyPresent = true;
628  break;
629  }
630 
631  case SPINEL_PROP_PHY_CHAN_SUPPORTED:
632  {

634 
635  opDataset.mChannelMask = 0;

638  {

642  }

710  spinel_datatype_vunpack_in_place(aBuffer, aLength, mPropertyFormat, mPropertyArgs);

714  }
715  }
716  else
717  {
718  if (aCommand == mExpectedCommand)
719  {
720  mError = OT_ERROR_NONE;
721  }
722  else
723  {
724  mError = OT_ERROR_DROP;
725  }
726  }
727  }
728  else
729  {

737 template <typename InterfaceType, typename ProcessContextType>

739  const uint8_t * aBuffer,
740  uint16_t aLength)
741 {

805  VerifyOrExit(unpacked > 0, error = OT_ERROR_PARSE);
806 
807  switch (logLevel)

810  case SPINEL_NCP_LOG_LEVEL_ALERT:
811  case SPINEL_NCP_LOG_LEVEL_CRIT:
812  otLogCritPlat("RCP => %s", logString);
813  break;
814 
815  case SPINEL_NCP_LOG_LEVEL_ERR:
816  case SPINEL_NCP_LOG_LEVEL_WARN:
817  otLogWarnPlat("RCP => %s", logString);
818  break;
819 
820  case SPINEL_NCP_LOG_LEVEL_NOTICE:
821  otLogNotePlat("RCP => %s", logString);
822  break;
823 
824  case SPINEL_NCP_LOG_LEVEL_INFO:
825  otLogInfoPlat("RCP => %s", logString);
826  break;
827 
828  case SPINEL_NCP_LOG_LEVEL_DEBUG:
829  default:
830  otLogDebgPlat("RCP => %s", logString);
831  break;
832  }
833  }
834 
835 exit:
836  LogIfFail("Failed to handle ValueIs", error);

839 template <typename InterfaceType, typename ProcessContextType>
840 otError RadioSpinel<InterfaceType, ProcessContextType>::ParseRadioFrame(otRadioFrame & aFrame,
841  const uint8_t *aBuffer,
842  uint16_t aLength)
843 {
844  otError error = OT_ERROR_NONE;
845  uint16_t flags = 0;
846  int8_t noiseFloor = -128;
847  spinel_size_t size = OT_RADIO_FRAME_MAX_SIZE;
848  unsigned int receiveError = 0;
849  spinel_ssize_t unpacked;
850 
851  unpacked = spinel_datatype_unpack_in_place(aBuffer, aLength,
852  SPINEL_DATATYPE_DATA_WLEN_S // Frame
853  SPINEL_DATATYPE_INT8_S // RSSI
854  SPINEL_DATATYPE_INT8_S // Noise Floor
855  SPINEL_DATATYPE_UINT16_S // Flags
856  SPINEL_DATATYPE_STRUCT_S( // PHY-data
857  SPINEL_DATATYPE_UINT8_S // 802.15.4 channel
858  SPINEL_DATATYPE_UINT8_S // 802.15.4 LQI
859  SPINEL_DATATYPE_UINT64_S // Timestamp (us).
860  ) SPINEL_DATATYPE_STRUCT_S( // Vendor-data
861  SPINEL_DATATYPE_UINT_PACKED_S // Receive error
862  ),
863  aFrame.mPsdu, &size, &aFrame.mInfo.mRxInfo.mRssi, &noiseFloor, &flags,
864  &aFrame.mChannel, &aFrame.mInfo.mRxInfo.mLqi,
865  &aFrame.mInfo.mRxInfo.mTimestamp, &receiveError);
866 
867  VerifyOrExit(unpacked > 0, error = OT_ERROR_PARSE);
868

872 
873  aFrame.mInfo.mRxInfo.mAckedWithFramePending = ((flags & SPINEL_MD_FLAG_ACKED_FP) != 0);
874  }
875  else if (receiveError < OT_NUM_ERRORS)
876  {
877  error = static_cast<otError>(receiveError);
878  }
879  else
880  {
881  error = OT_ERROR_PARSE;
882  }
883 
884 exit:
885  LogIfFail("Handle radio frame failed", error);
