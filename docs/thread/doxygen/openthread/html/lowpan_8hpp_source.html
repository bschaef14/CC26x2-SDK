175

178 
179  exit:

181  }

195  otError Write(const Message &aMessage, uint8_t aLength)
196  {
197  otError error = OT_ERROR_NONE;
198  int rval;
199 
200  OT_UNUSED_VARIABLE(rval);
201 
202  VerifyOrExit(CanWrite(aLength), error = OT_ERROR_NO_BUFS);
203 
204  rval = aMessage.Read(aMessage.GetOffset(), aLength, mWritePointer);
205  OT_ASSERT(rval == aLength);
206 
207  mWritePointer += aLength;
208 
209  exit:
210  return error;
211  }
